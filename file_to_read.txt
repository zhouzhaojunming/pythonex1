# Define the input and output file namesinput_file_name = "file_to_read.txt"output_file_name = "result.txt"# Initialize a dictionary to store word countsword_counts = {}# Open the input file for readingwith open(input_file_name, "r") as input_file:    # Read the entire content of the file    text = input_file.read()    # Split the text into words and count the occurrences of "terrible"    words = text.split()    for word in words:        word = word.strip(".,!?():;\"'")        if word.lower() == "terrible":            if word not in word_counts:                word_counts[word] = 1            else:                word_counts[word] += 1# Initialize a variable to keep track of whether the current occurrence is even or oddeven_occurrence = True# Replace even and odd occurrences of "terrible" with "pathetic" and "marvelous"modified_text = ""words = text.split()for word in words:    stripped_word = word.strip(".,!?():;\"'")    if stripped_word.lower() == "terrible":        if even_occurrence:            modified_text += "pathetic "        else:            modified_text += "marvelous "        even_occurrence = not even_occurrence    else:        modified_text += word + " "# Write the modified text to the output filewith open(output_file_name, "w") as output_file:    output_file.write(modified_text)# Display the total count of "terrible" occurrencestotal_terrible_count = sum(word_counts.values())print(f"Total 'terrible' count: {total_terrible_count}")# Print a message to indicate the task is completeprint("Replacement and writing to result.txt complete.")
